ğŸ” Basic Usage
find .                               # List all files/dirs recursively
find /path                           # Search in specific path
find . -name "file.txt"              # Find by exact name
find . -iname "*.txt"                # Case-insensitive name search
find . -name "*.log" -o -name "*.tmp" # OR conditions

ğŸ“ File Type Filters
find . -type f                       # Files only
find . -type d                       # Directories only
find . -type l                       # Symbolic links only
find . -type f -name "*.sh"          # Shell scripts
find . -type d -name "src"           # Directories named src

ğŸ“ Size Filters
find . -size +100M                   # Files larger than 100MB
find . -size -1M                     # Files smaller than 1MB
find . -size 50k                     # Files exactly 50KB
find . -empty                        # Empty files/directories
find . -type f -size +1G             # Files over 1GB

ğŸ“… Time-based Searches
find . -mtime -7                     # Modified in last 7 days
find . -mtime +30                    # Modified more than 30 days ago
find . -atime -1                     # Accessed in last 24 hours
find . -ctime -7                     # Changed in last 7 days
find . -mmin -60                     # Modified in last 60 minutes
find . -newer file.txt               # Newer than file.txt

ğŸ” Permission and Ownership
find . -perm 644                     # Exact permissions
find . -perm -644                    # At least these permissions
find . -perm /644                    # Any of these permissions
find . -user username                # Files owned by user
find . -group groupname              # Files owned by group
find . -nouser                       # Files with no owner
find . -executable                   # Executable files

ğŸ§¹ Excluding Directories
find . -name "*.js" -not -path "*/node_modules/*"
find . -path ./dir -prune -o -print  # Exclude ./dir
find . -type d -name .git -prune -o -type f -print
find . -name "*.txt" ! -path "*/tmp/*"

ğŸ”— Following Symlinks
find -L . -type f                    # Follow symbolic links
find -P . -type l                    # Don't follow symlinks (default)
find . -type l -ls                   # List symbolic links

âš¡ Executing Commands
find . -name "*.log" -delete         # Delete matching files
find . -type f -exec chmod 644 {} \; # Change permissions
find . -name "*.bak" -exec rm {} +   # Delete (faster with +)
find . -name "*.txt" -exec cat {} \; # Cat all txt files
find . -type f -exec grep -l "text" {} \;  # Search in files

ğŸ§ª Advanced Filters
find . -regex ".*\.py$"              # Regex pattern
find . -maxdepth 2 -name "*.js"      # Limit depth to 2 levels
find . -mindepth 2 -name "*.md"      # Start from depth 2
find . -name "*.tmp" -ls             # List in ls format
find . -printf "%p %s\n"             # Custom output format

ğŸ”„ Combining Conditions
find . -type f -name "*.log" -size +10M     # AND conditions
find . \( -name "*.txt" -o -name "*.md" \)  # Grouped OR
find . -name "*.js" -not -name "*.min.js"   # Exclusion
find . -type f -size +1M -mtime -7          # Large recent files

ğŸ“Š Statistics and Counting
find . -type f | wc -l               # Count files
find . -type f -printf "." | wc -c   # Count files (faster)
find . -type f -exec du -ch {} + | grep total  # Total size
find . -name "*.js" -exec wc -l {} + # Count lines in files

ğŸ› ï¸ Practical Examples

ğŸ” Name Patterns
